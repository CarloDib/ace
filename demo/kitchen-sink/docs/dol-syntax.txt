Document           ::= LibraryItem*
LibraryItem        ::= OMSDefinition | Qualification
Qualification      ::= 'logic' IRI
OMSDefinition      ::= OMSkeyword IRI '=' OMS 'end'
OMSkeyword         ::= 'ontology'
                     | 'onto'
                     | 'specification'
                     | 'spec'
                     | 'model'
                     | 'oms'
BasicOMS           ::= <see OWL and CLIF syntax>
OMS                ::= '{' BasicOMS '}' 
                     | IRI
                     | OMS 'and' OMS
                     | OMS 'then' OMS
OMSRef             ::= IRI
OMSName            ::= IRI
IRI ::= <sequence of letters>
